  studentId: /^[A-Za-z0-9]{6,12}$/,
  // School code format (uppercase alphanumeric, 4-8 characters)
  schoolCode: /^[A-Z0-9]{4,8}$/,

  orderCode: /^HSV-\d{8}-\d{4}$/,
