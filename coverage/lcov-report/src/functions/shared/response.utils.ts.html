/**
 * Create standardized success response
 */
export function createSuccessResponse<T>(
  data: T,
  context: Context,
  statusCode: number = 200
): APIGatewayProxyResult {
  const response: APIResponse<T> = {
    success: true,
    data,
    meta: {
      timestamp: new Date().toISOString(),
      correlationId: context.awsRequestId,
      version: '1.0.0',
      requestId: context.awsRequestId,
    },
  };
 
  return {
    statusCode,
    headers: {
      'Content-Type': 'application/json',
      'X-Correlation-ID': context.awsRequestId,
      'X-Request-ID': context.awsRequestId,
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET,POST,PUT,DELETE,OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type,Authorization,X-Amz-Date,X-Api-Key,X-Amz-Security-Token',
    },
    body: JSON.stringify(response),
  };
}

/**
 * Handle errors with proper logging and response formatting
 */
export function handleError(
  error: Error,
  statusCode: number = 500,
  context: Context,
  details?: any
): APIGatewayProxyResult {
  const logger = LoggerService;
 
  // Log error details
  logger.error('Lambda function error', error, {
    requestId: context.awsRequestId,
    functionName: context.functionName,
    statusCode,
    details,
  });
 
  // Determine error message based on environment
  const isProduction = process.env.NODE_ENV === 'production';
  const errorMessage = isProduction && statusCode >= 500 
    ? 'Internal server error occurred' 
    : error.message;
 
  // Map common error patterns to appropriate status codes
  const mappedStatusCode = mapErrorToStatusCode(error, statusCode);
 
  return createErrorResponse(
    errorMessage,
    mappedStatusCode,
    context,
    isProduction && statusCode >= 500 ? undefined : details,
    getErrorCode(error)
  );
}
 
/**
 * Create paginated response
 */
export function createPaginatedResponse<T>(
  items: T[],
  total: number,
  page: number,
  limit: number,
  context: Context,
  additional?: any
): APIGatewayProxyResult {
  const totalPages = Math.ceil(total / limit);
  
  const response = {
    items,
    pagination: {
      page,
      limit,
      total,
      totalPages,
      hasNext: page < totalPages,
      hasPrev: page > 1,
      nextPage: page < totalPages ? page + 1 : null,
      prevPage: page > 1 ? page - 1 : null,
    },
    ...additional,
  };
 
  return createSuccessResponse(response, context);
}

  const response = {

    uptime: process.uptime(),
    environment: process.env.NODE_ENV || 'development',
